# Turn end "--" into 0 and delete every cell that still has "--" to get rid of dates that are too arbitrary (month or year is unknown)

lst <- lst |>
  mutate(
    across(-c(recidnum, TB_yn), ~gsub("--$", "00", .)),
    across(-c(recidnum, TB_yn), ~if_else(grepl("--", .x), NA, .x)),
    across(-c(recidnum, TB_yn), ~ifelse(!is.na(.x), substr(.x, 1, 4), .x))
  ) |>
  filter(if_any(-c(recidnum, TB_yn), ~ !is.na(.x))) |>
  rowwise() |>
  mutate(
    first_lst = min(c_across(-c(recidnum,TB_yn)), na.rm = TRUE)
  ) |>
  ungroup()

As you can see, the tb vs. non-tb survival curves are significantly different (P < 0.0001 for log rank test).

The log rank test requires that: - the censoring patterns are the same - the hazard functions for the two groups are proportional

We want to see if their is any causal relationship between the diagnosis of tb. To do this, we will match TB individuals to non-TB individuals with as many of the same characteristics as possible. The characteristics we will try to match are:

  • Birth Year

  • US State of enlistment

# Dataframe for matching from the previous time_since_tb dataframe

match_full <- surv_tb |>
  select(
    recidnum,
    tb_first_date, age_tb, age_death,
    BYR, DYR, initstat, TB_yn
  ) |>
  filter(
    if_all(c(BYR), is.finite)
  ) |>
  mutate(
    TB_yn = as.numeric(as.character(TB_yn))
  ) |>
  drop_na(initstat)


match_full <- matchit(TB_yn ~ BYR + initstat,
                      data = match_full,
                      distance = "glm",
                      link = "probit")

matched_df <- match.data(match_full)


# Add constant status for lack of censoring
matched_df$status <- 1

# Create index date for non-TB
matched_df <- matched_df |>
  group_by(subclass) |>
  mutate(
    index_date = max(tb_first_date, na.rm = T)
  ) |>
  ungroup() |>
  mutate(
    index_date = if_else(is.infinite(index_date), NA_real_, index_date)
  ) |>
  drop_na(index_date) |>
  mutate(
    survival_time = DYR - index_date
  ) |>
  drop_na(survival_time) |>
  filter(survival_time >= 0)
  
# Final formatting of the matched data
plot_df <- matched_df |>
  mutate(
    event = 1
  ) |>
  mutate(TB_yn = factor(TB_yn, 
                        levels = c(0, 1), 
                        labels = c("Non-TB", "TB")))

# Fit km model
km_matched <- survfit(Surv(survival_time, event) ~ TB_yn, data = plot_df)

matched_p_final <- ggsurvplot(
  km_matched, 
  data = plot_df,
  conf.int = TRUE,       
  risk.table = FALSE,    
  pval = TRUE,  
  palette = c("#E7B800", "#2E9FDF"), 
  xlab = "Years of Life Remaining after Index Date",

)
